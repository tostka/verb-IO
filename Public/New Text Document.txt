#*------v Function touch-File v------
function touch-File {
    <#
    .SYNOPSIS
    touch-File - Create an empty file (approximates *nix touch command)
    .NOTES
    Version     : 1.0.0
    Author      : LittleBoyLost
    Website     :	https://superuser.com/users/210235/littleboylost
    Twitter     :	
    CreatedDate : 2013-03-25
    FileName    : touch-File.ps1
    License     : 
    Copyright   : 
    Github      : 
    Tags        : Powershell,File,Filesystem
    REVISIONS
    * 2:03 PM 3/27/2020 touch-File:rpl code with param validat, simplified, added CBH, cited author & source
    * 3/25/2013 posted version
    .DESCRIPTION
    touch-File - Create an empty file (approximates *nix touch command)
    .PARAMETER  File
    File [-file c:\path-to\file.ext]
    .EXAMPLE
    .\touch-File.ps1 c:\somefile.txt
    Create an empty file (approximates *nix touch command)
    .LINK
    https://superuser.com/questions/502374/equivalent-of-linux-touch-to-create-an-empty-file-with-powershell
    #>
    Param([Parameter(Position=0,Mandatory=$True,ValueFromPipeline=$true,ValueFromPipelineByPropertyName=$true,HelpMessage="File [-file c:\path-to\file.ext]")]
    [ValidateNotNullOrEmpty()]$File) ;
    if(Test-Path $file){(Get-ChildItem $file).LastWriteTime = Get-Date} else {echo $null > $file} ;
} ; #*------^ END Function touch-File  ^------
